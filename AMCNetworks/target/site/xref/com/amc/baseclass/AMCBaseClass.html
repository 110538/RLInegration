277 		String[][] excelData = null;
278 		try {
279 			int colCount = 0, rowCount = 0;
280 			sheet = workbook.getSheet(sheetName);
281 			colCount = sheet.getRow(0).getPhysicalNumberOfCells();
282 			rowCount = sheet.getPhysicalNumberOfRows();
283 			excelData = new String[rowCount][colCount];
284 			for (int Nrow = 0; Nrow < rowCount; Nrow++) {
285 				sheet.getRow(Nrow);
286 				for (int Ncolumn = 0; Ncolumn < colCount; Ncolumn++) {
287 					cell = sheet.getRow(Nrow).getCell(Ncolumn);
288 					DataFormatter df = new DataFormatter();
289 					excelData[Nrow][Ncolumn] = df.formatCellValue(cell);
290 				}
291 			}
292 		} catch (Exception e) {
293 			System.out.println(e.getMessage());
294 		}
295 		return excelData;
296 	}

337 		switch (browserType) {
338 		case "firefox":
339 			driver = initalizeFirefoxDriver(appURL);
340 			break;
341 		case "chrome":
342 			driver = initalizeChromeDriver(appURL);

344 		case "ie":
345 			driver = initalizeIEDriver(appURL);
346 			break;
347 		case "FireFoxForBrowserStack":
348 			driver = initalizeFirefoxDriverForBrowserStack(appURL);
349 			break;
350 		case "ChromeForBrowserStack":
351 			driver = initalizeChromeDriverForBrowserStack(appURL);
352 			break;
353 		default:
354 			initalizeFirefoxDriver(appURL);
355 		}
